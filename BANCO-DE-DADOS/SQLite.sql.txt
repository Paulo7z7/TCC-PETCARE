CREATE database TCC;
USE TCC;
  
CREATE TABLE CLIENTE(
NOME VARCHAR(50) NOT NULL,
CPF VARCHAR(11) NOT NULL,
RG VARCHAR(9) NOT NULL,
SEXO VARCHAR(30),
DTNASCIMENTO DATE NOT NULL,
ESTADO VARCHAR(30) NOT NULL,
CIDADE VARCHAR(30) NOT NULL,
CEP INT NOT NULL,
BAIRRO VARCHAR(30) NOT NULL,
RUA VARCHAR(30) NOT NULL,
NUMERO VARCHAR(15) NOT NULL,
TELEFONE VARCHAR(11) NOT NULL,
EMAIL VARCHAR(50) NOT NULL,
CODIGOIDCLIEN INT NOT NULL,
PRIMARY KEY (CODIGOIDCLIEN)
);

CREATE TABLE PET(
NOME VARCHAR(50) NOT NULL,
ESPECIE VARCHAR(50) NOT NULL,
RAÇA VARCHAR(50) NOT NULL,
PORTE VARCHAR(30) NOT NULL,
SEXO VARCHAR(30),
IDADE INT NOT NULL,
OBSERVAÇÃO VARCHAR(100) NOT NULL,
CODIGOIDCLIEN INT NOT NULL,
CODIGOIDPET INT NOT NULL,
PRIMARY KEY (CODIGOIDPET),
FOREIGN KEY (CODIGOIDCLIEN) REFERENCES CLIENTE (CODIGOIDCLIEN)
);

CREATE TABLE EMPRESA(
NOME VARCHAR(50) NOT NULL,
CNPJ VARCHAR(30) NOT NULL,
IE VARCHAR(30) NOT NULL,
ESTADO VARCHAR(30) NOT NULL,
CIDADE VARCHAR(30) NOT NULL,
CEP INT NOT NULL,
BAIRRO VARCHAR(30) NOT NULL,
RUA VARCHAR(30) NOT NULL,
NUMERO VARCHAR(15) NOT NULL,
TELEFONE VARCHAR(11) NOT NULL,
EMAIL VARCHAR(50) NOT NULL,
CODIGOIDEMPRE INT NOT NULL,
PRIMARY KEY (CODIGOIDEMPRE)
);

CREATE TABLE FUNCIONARIO(
NOME VARCHAR(50) NOT NULL,
CPF VARCHAR(11) NOT NULL,
RG VARCHAR(9) NOT NULL,
SEXO VARCHAR(30),
DTNASCIMENTO DATE NOT NULL,
ESTADO VARCHAR(30) NOT NULL,
CIDADE VARCHAR(30) NOT NULL,
CEP INT NOT NULL,
BAIRRO VARCHAR(30) NOT NULL,
RUA VARCHAR(30) NOT NULL,
NUMERO VARCHAR(15) NOT NULL,
TELEFONE VARCHAR(11) NOT NULL,
SENHA VARCHAR(50) NOT NULL,
CODIGOIDFUNC INT NOT NULL,
CODIGOIDEMPRE INT NOT NULL,
PRIMARY KEY (CODIGOIDFUN),
FOREIGN KEY (CODIGOIDEMPRE) REFERENCES EMPRESA (CODIGOIDEMPRE)
);
